// state structure 

const DISPLAY_TODO_TYPE = {
    ALL:"ALL",
    ACTIVE:"ACTIVE",
    COMPLETED:"COMPLETED"
}

todo == stores all the todo list 
displayTodoType = DISPLAY_TODO_TYPE.ALL;

onButtonClick => update displayTodoType

const filter = (todo)=>{
    if(displayTodoType==ALL) return true;
    else if(displayTodoType==ACTIVE) return !todo.isCompleted
    else if(displayTodoType===COMPLETED) return todo.isCompleted
}

const tobedisplayedTodo = todo.filter(filter)

render tobedisplayedTodo;
